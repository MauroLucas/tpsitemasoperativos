#include <stdio.h>
#include <mpi.h>

int main(int argc, char *argv[])
{
    int rank, size;
    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int x = 0;
    if (rank == 0) {
        for (x = 1; x < 1001; x++) {
            printf("Soy el core nro. %d de %d: x = %d\n", rank, size, x);
        }
    }

    MPI_Finalize();
    return 0;
}
