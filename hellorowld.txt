#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "mpi.h"

#define MAX_MSG_LENGTH 50
#define MAX_MSG_COUNT 1000

char messages[MAX_MSG_COUNT][MAX_MSG_LENGTH];

int main(int argc, char *argv[])
{
    int rank, size;
    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    int x;
    for (x = 0; x < MAX_MSG_COUNT; x++){
        sprintf(messages[x], "Soy el core nro. %d de %d\n", rank, size);
        printf("%s", messages[x]); // Imprime el mensaje
    }

    char filename[20];
    sprintf(filename, "output_rank_%d.csv", rank);
    FILE *f = fopen(filename, "w");
    if (f == NULL)
    {
        printf("Error abriendo el archivo!\n");
        MPI_Finalize();
        return 1;
    }

    for (x = 0; x < MAX_MSG_COUNT; x++){
        fprintf(f, "%s", messages[x]); // Escribe el mensaje en el archivo
    }

    fclose(f);
    MPI_Finalize();
    return 0;
}
