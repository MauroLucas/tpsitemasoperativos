#include <stdio.h>
#include <mpi.h>

int main(int argc, char *argv[])
{
    int rank, size;
    MPI_Init(&argc, &argv);

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    // Abrir el archivo en modo escritura (cada proceso escribir√° en el mismo archivo)
    MPI_File fh;
    MPI_File_open(MPI_COMM_WORLD, "output.csv", MPI_MODE_WRONLY | MPI_MODE_CREATE, MPI_INFO_NULL, &fh);

    // Escribir los datos en el archivo CSV
    char buffer[100];
    sprintf(buffer, "Soy el core nro. %d de %d\n", rank, size);

    MPI_File_write(fh, buffer, strlen(buffer), MPI_CHAR, MPI_STATUS_IGNORE);

    // Cerrar el archivo
    MPI_File_close(&fh);

    MPI_Finalize();
    return 0;
}
